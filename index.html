async function loadData() { const { data: { user } } = await sb.auth.getUser(); if (!user) return; const today = new Date().toISOString().split('T')[0]; const { data, error } = await sb .from('entries') .select('*') .eq('user_id', user.id) .eq('date', today); if (error) { console.error(error); return; } let tIn = 0, tSport = 0, tP = 0, tG = 0, tL = 0; const entries = (data || []).sort((a, b) => b.id - a.id); entries.forEach(e => { if (e.type === 'food') { tIn += e.val; tP += e.prot || 0; tG += e.gluc || 0; tL += e.lip || 0; } else { tSport += e.val; } }); const bmr = calcBMR(userWeight, userHeight, userAge, userSexe); const totalOut = bmr + tSport; const remaining = userGoal + tSport - tIn; const pct = Math.min(100, Math.round((tIn / userGoal) * 100)); // Mise à jour UI document.getElementById('kcal-left').textContent = Math.max(0, remaining); document.getElementById('stat-in').textContent = tIn; document.getElementById('stat-out').textContent = totalOut; document.getElementById('stat-bmr').textContent = bmr; document.getElementById('stat-sport').textContent = tSport; document.getElementById('total-p').textContent = Math.round(tP) + ' g'; document.getElementById('total-g').textContent = Math.round(tG) + ' g'; document.getElementById('total-l').textContent = Math.round(tL) + ' g'; document.getElementById('progress-fill').style.width = pct + '%'; document.getElementById('progress-eaten').textContent = tIn + ' mangé'; document.getElementById('progress-goal').textContent = userGoal + ' objectif'; // Couleur barre const bar = document.getElementById('progress-fill'); if (pct >= 95) { bar.style.background = 'linear-gradient(90deg,#ef4444,#f97316)'; } else if (pct >= 75) { bar.style.background = 'linear-gradient(90deg,#fbbf24,#f97316)'; } else { bar.style.background = 'linear-gradient(90deg,var(--accent),var(--accent2))'; } // Couleur kcal restantes const kcalEl = document.getElementById('kcal-left'); if (remaining < 200) { kcalEl.style.background = 'linear-gradient(135deg,#ef4444,#f87171)'; } else if (remaining < 500) { kcalEl.style.background = 'linear-gradient(135deg,#fbbf24,#f97316)'; } else { kcalEl.style.background = 'linear-gradient(135deg,#fff 30%,#7a8098)'; } kcalEl.style.webkitBackgroundClip = 'text'; kcalEl.style.webkitTextFillColor = 'transparent'; // Liste du jour const listEl = document.getElementById('daily-list'); const titleEl = document.getElementById('list-title'); if (!entries.length) { titleEl.style.display = 'none'; listEl.innerHTML = `
🥗
Aucun aliment enregistré aujourd'hui
`; return; } titleEl.style.display = 'block'; const ORDER = ['Petit-déjeuner', 'Déjeuner', 'Dîner', 'Snack', 'Boissons', 'Autre', 'Sport']; const ICONS = { 'Petit-déjeuner': '🥐', 'Déjeuner': '🍽️', 'Dîner': '🌙', 'Snack': '🍎', 'Boissons': '🥤', 'Autre': '➕', 'Sport': '🔥' }; const groups = {}; ORDER.forEach(cat => groups[cat] = []); entries.forEach(e => { if (groups[e.category] !== undefined) { groups[e.category].push(e); } else { groups['Autre'].push(e); } }); let html = ''; ORDER.forEach(cat => { const items = groups[cat]; if (!items.length) return; const subtotal = items.reduce((sum, e) => sum + e.val, 0); const isSport = cat === 'Sport'; html += `
${ICONS[cat]} ${cat}
${isSport ? '−' : '+'}${subtotal} kcal
${items.map(e => `
${e.desc}
${e.type === 'burn' ? '−' : '+'}${e.val} kcal
`).join('')}
`; }); listEl.innerHTML = html; }
